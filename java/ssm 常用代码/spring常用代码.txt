1.注入servlet(response不能注入)
@Autowired
private ServletContext servletContext;
	
@Autowired
private HttpServletRequest request;

2.获取项目绝对路径（tomcat目录下）
String path=servletContext.getRealPath("/");

3.获取请求相关信息
StringBuffer url=request.getRequestURL();    //http地址
String uri=request.getRequestURI();		//项目路径
String referer=request.getHeader("referer");	//来源

4.上传文件
a.引入commons-fileupload jar
b.配置 
<bean id="multipartResolver" class="org.springframework.web.multipart.commons.CommonsMultipartResolver">
    	<property name="defaultEncoding" value="UTF-8"></property>
    	<!-- 20M -->
    	<property name="maxUploadSize" value="20971520"></property>
</bean>
c.前端 form 属性 enctype="multipart/form-data"
d.后台接收参数 @RequestParam("images") MultipartFile file

5.中文乱码问题
a.后台接收参数乱码
<!-- 后台接收参数乱码处理 只能适用post请求 -->
<filter>
    	<description>字符编码过滤器</description>
        <filter-name>encodingFilter</filter-name>
        <filter-class>org.springframework.web.filter.CharacterEncodingFilter</filter-class>
        <init-param>
            <param-name>encoding</param-name>
            <param-value>UTF-8</param-value>
        </init-param>
        <init-param>
            <param-name>forceEncoding</param-name>
            <param-value>true</param-value>
        </init-param>
</filter>
<filter-mapping>
<filter-name>encodingFilter</filter-name>
     <url-pattern>/*</url-pattern>
</filter-mapping>
